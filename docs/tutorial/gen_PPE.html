# gen pseudoproxy

SNR = 1

pp = np.copy(tas_sub)
nt, nlat, nlon = np.shape(pp)

k = 0
for i in range(nlat):
    for j in range(nlon):
        tas_std = np.std(tas_sub[:,i,j])
        noise_std = tas_std / SNR

        np.random.seed(k)
        pp[:,i,j] += np.random.normal(0, noise_std, size=nt)

        k += 1
